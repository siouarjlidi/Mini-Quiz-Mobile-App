  Widget _buildLeaderboardTab(String difficulty) {
    return StreamBuilder<QuerySnapshot>(
      stream: FirebaseFirestore.instance
          .collection('scores')
          .where('difficulty', isEqualTo: difficulty)
          .snapshots(),
      builder: (context, snapshot) {
        if (snapshot.connectionState == ConnectionState.waiting) {
          return const Center(child: CircularProgressIndicator());
        }

        if (!snapshot.hasData || snapshot.data!.docs.isEmpty) {
          return Center(
            child: Column(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                Icon(
                  Icons.leaderboard_outlined,
                  size: 80,
                  color: Colors.grey.shade300,
                ),
                const SizedBox(height: 20),
                Text(
                  'Aucun score pour $difficulty',
                  style: TextStyle(
                    fontSize: 18,
                    color: Colors.grey.shade600,
                  ),
                ),
              ],
            ),
          );
        }

        List<QueryDocumentSnapshot> scores = snapshot.data!.docs;

        final Map<String, QueryDocumentSnapshot> bestScores = {};

        for (final score in scores) {
          final data = score.data() as Map<String, dynamic>;
          final userName = data['userName']?.toString().trim() ?? 'Anonyme';
          final percentage = (data['percentage'] ?? 0) as int;

          // Si l'utilisateur n'existe pas encore OU si ce score est meilleur
          if (!bestScores.containsKey(userName) ||
              percentage > ((bestScores[userName]!.data() as Map<String, dynamic>)['percentage'] ?? 0)) {
            bestScores[userName] = score;
          }
        }

        // Convertir en liste et trier par pourcentage
        final uniqueScores = bestScores.values.toList();
        uniqueScores.sort((a, b) {
          final aData = a.data() as Map<String, dynamic>;
          final bData = b.data() as Map<String, dynamic>;
          final aPercentage = (aData['percentage'] ?? 0) as int;
          final bPercentage = (bData['percentage'] ?? 0) as int;
          return bPercentage.compareTo(aPercentage);
        });

        return ListView.builder(
          padding: const EdgeInsets.all(16),
          itemCount: uniqueScores.length,
          itemBuilder: (context, index) {
            final score = uniqueScores[index];
            final data = score.data() as Map<String, dynamic>;
            final userName = data['userName']?.toString().trim() ?? 'Anonyme';
            final scoreValue = (data['score'] ?? 0) as int;
            final totalQuestions = (data['totalQuestions'] ?? 1) as int;
            final percentage = (data['percentage'] ?? 0) as int;
            final dateStr = data['date'];

            // Formater la date
            String formattedDate = 'Aujourd\'hui';
            if (dateStr != null) {
              try {
                final date = DateTime.parse(dateStr as String).toLocal();
                final now = DateTime.now();
                final difference = now.difference(date);

                if (difference.inDays == 0) {
                  formattedDate = 'Aujourd\'hui';
                } else if (difference.inDays == 1) {
                  formattedDate = 'Hier';
                } else if (difference.inDays < 7) {
                  formattedDate = 'Il y a ${difference.inDays} jours';
                } else {
                  formattedDate = '${date.day}/${date.month}/${date.year}';
                }
              } catch (e) {
                formattedDate = 'Date inconnue';
              }
            }

            return Card(
              margin: const EdgeInsets.only(bottom: 12),
              child: ListTile(
                leading: CircleAvatar(
                  backgroundColor: index < 3 ? _getRankColor(index) : Colors.grey.shade200,
                  child: Text(
                    '${index + 1}',
                    style: TextStyle(
                      color: index < 3 ? Colors.white : Colors.grey.shade700,
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                ),
                title: Text(
                  userName,
                  style: const TextStyle(fontWeight: FontWeight.bold),
                ),
                subtitle: Text('$scoreValue/$totalQuestions ‚Ä¢ $formattedDate'),
                trailing: Chip(
                  label: Text(
                    '$percentage%',
                    style: const TextStyle(
                      color: Colors.white,
                      fontWeight: FontWeight.bold,
                    ),
                  ),
                  backgroundColor: _getPercentageColor(percentage),
                ),
              ),
            );
          },
        );
      },
    );
  }

  Color _getRankColor(int index) {
    switch (index) {
      case 0: return Colors.amber;
      case 1: return Colors.grey;
      case 2: return Colors.brown;
      default: return Colors.blue;
    }
  }

  Color _getPercentageColor(int percentage) {
    if (percentage >= 80) return Colors.green;
    if (percentage >= 60) return Colors.orange;
    return Colors.red;
  }
}

  @override
  void dispose() {
    _timer?.cancel();
    _nameController.dispose();
    super.dispose();
  }

  void _startTimer() {
    _timer?.cancel();
    _timeLeft = 30;
    _timer = Timer.periodic(const Duration(seconds: 1), (timer) {
      if (_timeLeft > 0) {
        setState(() => _timeLeft--);
      } else {
        _timer?.cancel();
        _showCorrectionAndNext(false, '‚è∞ Temps √©coul√© !');
      }
    });
  }

  void _showCorrectionAndNext(bool isCorrect, String correction) {
    setState(() {
      _showCorrection = true;
      _correctionText = correction;
      _isCorrect = isCorrect;
    });

    if (isCorrect) {
      _score++;
    }

    Timer(const Duration(seconds: 3), () {
      setState(() {
        _showCorrection = false;
      });
      _nextQuestion();
    });
  }

  void _nextQuestion() {
    final questions = _questions[_difficulty]!;
    if (_currentQuestion < questions.length - 1) {
      setState(() => _currentQuestion++);
      _startTimer();
    } else {
      _finishQuiz();
    }
  }

  Future<void> _saveScoreToFirebase() async {
    if (_userName == null) return;

    try {
      // Nettoyer le nom de l'utilisateur
      final cleanUserName = _userName!.trim();

      await _firestore.collection('scores').add({
        'userName': cleanUserName,
        'score': _score,
        'difficulty': _difficulty,
        'date': DateTime.now().toIso8601String(),
        'totalQuestions': _questions[_difficulty]!.length,
        'percentage': (_score / _questions[_difficulty]!.length * 100).round(),
      });
      print('‚úÖ Score sauvegard√© pour $cleanUserName');
    } catch (e) {
      print('‚ö†Ô∏è Erreur sauvegarde score: $e');
    }
  }

  void _startQuiz() {
    if (_nameController.text.isNotEmpty) {
      final name = _nameController.text.trim();
      setState(() {
        _userName = name;
        _score = 0;
        _currentQuestion = 0;
      });
      _startTimer();
    }
  }

  void _answerQuestion(int selectedIndex) {
    _timer?.cancel();

    final questions = _questions[_difficulty]!;
    final question = questions[_currentQuestion];
    bool isCorrect = selectedIndex == question['answer'];

    _showCorrectionAndNext(isCorrect, question['explication'] as String);
  }

  Widget _buildDifficultyCard(String level, String title, Color color, IconData icon) {
    return GestureDetector(
      onTap: () => setState(() => _difficulty = level),
      child: Container(
        width: 100,
        padding: const EdgeInsets.all(16),
        decoration: BoxDecoration(
          color: _difficulty == level ? color.withOpacity(0.15) : Colors.white,
          borderRadius: BorderRadius.circular(12),
          border: Border.all(
            color: _difficulty == level ? color : Colors.grey.shade300,
            width: 2,
          ),
          boxShadow: [
            BoxShadow(
              color: Colors.black.withOpacity(0.05),
              blurRadius: 10,
              spreadRadius: 2,
            ),
          ],
        ),
        child: Column(
          children: [
            Icon(
              icon,
              color: _difficulty == level ? color : Colors.grey,
              size: 36,
            ),
            const SizedBox(height: 8),
            Text(
              title,
              style: TextStyle(
                color: _difficulty == level ? color : Colors.grey,
                fontWeight: FontWeight.bold,
                fontSize: 12,
              ),
            ),
          ],
        ),
      ),
    );
  }

  Color _getDifficultyColor(String difficulty) {
    switch (difficulty) {
      case 'facile': return Colors.green;
      case 'moyen': return Colors.orange;
      case 'difficile': return Colors.red;
      default: return Colors.blue;
    }
  }

  Color _getTimeColor() {
    if (_timeLeft > 18) return Colors.green;
    if (_timeLeft > 9) return Colors.orange;
    return Colors.red;
  }

  IconData _getDifficultyIcon(String difficulty) {
    switch (difficulty) {
      case 'facile': return Icons.star_border;
      case 'moyen': return Icons.star_half;
      case 'difficile': return Icons.star;
      default: return Icons.star;
    }
  }
